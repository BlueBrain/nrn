79 /* solve_dd_clhs_tridiag uses Thomas Algorithm to solves a
80  * diagonally dominant tridiagonal matrix (Ax=b), where the
81  * triple (lower, diagonal and upper) are repeated to form A.
82  * N - length of the matrix
83  * l_diag - lower diagonal
84  * diag - diagonal
85  * u_diag - upper diagonal
86  * b - pointer to the RHS (length N)
87  * lbc_diag - the lower boundary diagonal (the first diagonal element)
88  * lbc_u_diag - the lower boundary upper diagonal (the first upper
89  * diagonal element)
90  * ubc_l_diag - the upper boundary lower diagonal (the last lower
91  * diagonal element)
92  * ubc_diag - the upper boundary diagonal (the last diagonal element)
93  * The solution (x) is stored in b.
94  * c - scratchpad array, N - 1 doubles long
95  */
96 static int solve_dd_clhs_tridiag(const int N, const double l_diag, const double diag,
97  const double u_diag, const double lbc_diag, const double lbc_u_diag,
98  const double ubc_l_diag, const double ubc_diag, double* const b, double* const c)
99 {
100  int i;
101 
102 
103  c[0] = lbc_u_diag/lbc_diag;
104  b[0] = b[0]/lbc_diag;
105 
106  for(i=1;i<N-1;i++)
107  {
108  c[i] = u_diag/(diag-l_diag*c[i-1]);
109  b[i] = (b[i]-l_diag*b[i-1])/(diag-l_diag*c[i-1]);
110  }
111  b[N-1] = (b[N-1]-ubc_l_diag*b[N-2])/(ubc_diag-ubc_l_diag*c[N-2]);
112 
113  /*back substitution*/
114  for(i=N-2;i>=0;i--)
115  {
116  b[i]=b[i]-c[i]*b[i+1];
117  }
118  return 0;
119 }

121 /* solve Ax=b where A is a diagonally dominant tridiagonal matrix
122  * N - length of the matrix
123  * l_diag - pointer to the lower diagonal (length N-1)
124  * diag - pointer to diagonal (length N)
125  * u_diag - pointer to the upper diagonal (length N-1)
126  * B - pointer to the RHS (length N)
127  * The solution (x) will be stored in B.
128  * l_diag, diag and u_diag are not changed.
129  * c - scratch pad, preallocated memory for (N-1) doubles
130  */
131 static int solve_dd_tridiag(int N, const double* l_diag, const double* diag,
132  const double* u_diag, double* b, double *c)
133 {
134  int i;
135 
136  c[0] = u_diag[0]/diag[0];
137  b[0] = b[0]/diag[0];
138 
139  for(i=1;i<N-1;i++)
140  {
141  c[i] = u_diag[i]/(diag[i]-l_diag[i-1]*c[i-1]);
142  b[i] = (b[i]-l_diag[i-1]*b[i-1])/(diag[i]-l_diag[i-1]*c[i-1]);
143  }
144  b[N-1] = (b[N-1]-l_diag[N-2]*b[N-2])/(diag[N-1]-l_diag[N-2]*c[N-2]);
145 
146 
147  /*back substitution*/
148  for(i=N-2;i>=0;i--)
149  {
150  b[i]=b[i]-c[i]*b[i+1];
151  }
152  return 0;
